\section{Chi-Squared Distribution}
% ============================================================================

\begin{frame}{The Chi-Squared Distribution}

\begin{block}{Definition}
The chi-squared statistic is used to test hypotheses about categorical data by comparing observed frequencies with expected frequencies.
\end{block}

\vspace{0.5cm}

\textbf{Formula:}
$$\chi^2 = \sum \frac{(O_i - E_i)^2}{E_i}$$

where $O_i$ = observed frequency, $E_i$ = expected frequency

\vspace{0.5cm}

\textbf{Properties:}
\begin{itemize}
\item Always non-negative ($\chi^2 \geq 0$)
\item Depends on degrees of freedom ($\nu$)
\item Right-skewed distribution
\item As $\nu$ increases, distribution approaches normal
\end{itemize}

\end{frame}

\begin{frame}{Hypotheses and Decision Rules}

\textbf{Hypothesis Testing Framework:}

\begin{itemize}
\item $H_0$: Null hypothesis (no association/good fit)
\item $H_1$: Alternative hypothesis (association exists/poor fit)
\end{itemize}

\vspace{0.5cm}

\textbf{Decision Rule:}
\begin{itemize}
\item Calculate test statistic: $\chi^2_{calc}$
\item Find critical value from tables: $\chi^2_{crit}(\alpha, \nu)$
\item If $\chi^2_{calc} > \chi^2_{crit}$: Reject $H_0$ (significant result)
\item If $\chi^2_{calc} \leq \chi^2_{crit}$: Do not reject $H_0$ (insufficient evidence)
\end{itemize}

\vspace{0.5cm}

\textbf{Conditions for Validity:}
\begin{itemize}
\item All expected frequencies $E_i \geq 5$
\item Independent observations
\item Random sampling
\end{itemize}

\end{frame}

\section{Goodness of Fit Test}
% ============================================================================

\begin{frame}{Goodness of Fit Test}

\begin{block}{Purpose}
Tests whether observed data follows a specified theoretical distribution or ratio.
\end{block}

\vspace{0.5cm}

\textbf{Degrees of Freedom:}
$$\nu = n - 1 - p$$
where $n$ = number of categories, $p$ = number of parameters estimated

\vspace{0.5cm}

\textbf{Hypotheses:}
\begin{itemize}
\item $H_0$: The data fits the specified distribution
\item $H_1$: The data does not fit the specified distribution
\end{itemize}

\vspace{0.5cm}

\textbf{Applications:}
\begin{itemize}
\item Testing fairness of dice/coins
\item Genetic ratio verification
\item Distribution fitting (uniform, binomial, etc.)
\end{itemize}

\end{frame}

\section{Test of Independence}
% ============================================================================

\begin{frame}{Test of Independence (Contingency Tables)}

\begin{block}{Purpose}
Tests whether two categorical variables are independent or associated.
\end{block}

\vspace{0.5cm}

\textbf{Degrees of Freedom:}
$$\nu = (r-1)(c-1)$$
where $r$ = number of rows, $c$ = number of columns

\vspace{0.5cm}

\textbf{Expected Frequency Formula:}
$$E_{ij} = \frac{(\text{Row}_i \text{ total}) \times (\text{Column}_j \text{ total})}{\text{Grand total}}$$

\vspace{0.5cm}

\textbf{Hypotheses:}
\begin{itemize}
\item $H_0$: The two variables are independent
\item $H_1$: The two variables are not independent (associated)
\end{itemize}

\end{frame}

\section{Summary}
% ============================================================================

\begin{frame}{Key Concepts Summary}

\begin{block}{Chi-Squared Test Statistic}
$$\chi^2 = \sum \frac{(O_i - E_i)^2}{E_i}$$
\end{block}

\begin{columns}

\column{0.5\textwidth}
\textbf{Goodness of Fit}
\begin{itemize}
\item Tests fit to distribution
\item $\nu = n - 1 - p$
\item Single variable
\item Compares observed vs theoretical
\end{itemize}

\column{0.5\textwidth}
\textbf{Test of Independence}
\begin{itemize}
\item Tests association
\item $\nu = (r-1)(c-1)$
\item Two variables
\item Uses contingency tables
\end{itemize}

\end{columns}

\vspace{0.5cm}

\textbf{Essential Conditions:}
\begin{itemize}
\item All expected frequencies $\geq 5$
\item Independent observations
\item Random sampling
\end{itemize}

\end{frame}
