\section{Z-Tests}

\begin{frame}{Problem 1: Quality Control — Battery Life}
  \begin{block}{Scenario}
    A battery manufacturer claims their AA batteries last 50 hours on average. 
    A quality control inspector tests 100 batteries and finds a mean life of 48.5 hours.
    The population standard deviation is known to be $\sigma = 6$ hours.
    
    \vspace{0.5em}
    At the 5\% significance level, test whether the actual mean battery life is less than claimed.
  \end{block}
  
  \textbf{Tasks:}
  \begin{enumerate}
    \item State $H_0$ and $H_a$
    \item Calculate the test statistic
    \item Find the critical value or p-value
    \item Make your decision
    \item Write a conclusion in context
  \end{enumerate}
\end{frame}

\begin{frame}{Problem 13: Education — Teaching Methods Comparison}

    Two teaching methods are compared. Method A is used with 35 students (mean score = 78, $s_1 = 12$). 
    Method B is used with 40 students (mean score = 82, $s_2 = 10$).
    
    \vspace{0.5em}
    At the 5\% level, test if there's a significant difference between the methods.
  \end{block}
  
  \textbf{Tasks:}
  \begin{enumerate}
    \item State $H_0: \mu_1 = \mu_2$ vs $H_a: \mu_1 \neq \mu_2$
    \item Calculate pooled standard deviation
    \item Compute t-statistic for two samples
    \item Find critical value
    \item Which method is better?
  \end{enumerate}
